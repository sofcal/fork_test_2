db.Bank.aggregate([
    { $match: { 'aggregatorName': 'yodlee' } },
    {
        $group: {
            _id: '$primaryCountry',
            banks: {
                $addToSet: {
                    bankId: '$_id',
                    name: '$name',
                    status: '$status'
                }
            }
        }
    }
], {
    allowDiskUse: true
})
